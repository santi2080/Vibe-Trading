   ```python
   def get_session_status(market_code: str, utc_dt: Optional[datetime] = None) -> MarketSessionStatus:
       """Return session status for a market at the given UTC time.
       
       Args:
           market_code: Market identifier (e.g., 'cn_stock', 'us_stock', 'us_futures')
           utc_dt: UTC datetime to check (defaults to now)
       
       Returns:
           MarketSessionStatus enum value
       """
       if utc_dt is None:
           utc_dt = datetime.now(timezone.utc)
       
       # Ensure UTC
       if utc_dt.tzinfo is None:
           utc_dt = utc_dt.replace(tzinfo=timezone.utc)
       
       code = market_code.lower()
       tz_name = MARKET_TZ.get(code)
       
       # Crypto and unknown markets: continuous
       if tz_name is None:
           return MarketSessionStatus.CONTINUOUS
       
       market_dt = utc_dt.astimezone(ZoneInfo(tz_name))
       t = market_dt.time()
       
       # A-share: 09:30-11:30 morning, 13:00-15:00 afternoon
       if code in ("cn_stock", "cn_stocks"):
           if time(9, 30) <= t < time(11, 30) or time(13, 0) <= t < time(15, 0):
               return MarketSessionStatus.REGULAR
           elif time(8, 30) <= t < time(9, 30):
               return MarketSessionStatus.PRE_MARKET
           elif time(15, 0) <= t < time(16, 0):
               return MarketSessionStatus.POST_MARKET
           return MarketSessionStatus.CLOSED
       
       # China futures: 09:00-10:15, 10:30-11:30, 13:30-15:00, 21:00-23:00
       if code in ("cn_futures",):
           if (time(9, 0) <= t < time(10, 15) or 
               time(10, 30) <= t < time(11, 30) or 
               time(13, 30) <= t < time(15, 0)):
               return MarketSessionStatus.REGULAR
           elif time(21, 0) <= t or t < time(23, 0):
               return MarketSessionStatus.REGULAR  # Night session
           elif time(8, 30) <= t < time(9, 0) or time(15, 0) <= t < time(21, 0):
               return MarketSessionStatus.POST_MARKET
           return MarketSessionStatus.CLOSED
       
       # US equity: 09:30-16:00 ET, pre 04:00-09:30, post 16:00-20:00
       if code in ("us_stock", "us_stocks"):
           if time(9, 30) <= t < time(16, 0):
               return MarketSessionStatus.REGULAR
           elif time(4, 0) <= t < time(9, 30):
               return MarketSessionStatus.PRE_MARKET
           elif time(16, 0) <= t < time(20, 0):
               return MarketSessionStatus.POST_MARKET
           return MarketSessionStatus.CLOSED
       
       # US futures: CME Globex 17:00-16:00 CT (cross-midnight)
       # 17:00-23:59 previous day CT to 00:00-16:00 current day CT
       if code in ("us_futures",):
           # Electronic session covers 17:00-16:00 (next day)
           if time(17, 0) <= t or t < time(16, 0):
               return MarketSessionStatus.REGULAR
           elif time(16, 0) <= t < time(17, 0):
               return MarketSessionStatus.POST_MARKET
           return MarketSessionStatus.CLOSED
       
       # HK stock: 09:30-12:00, 13:00-16:00 HKT
       if code in ("hk_stock", "hk_stocks"):
           if time(9, 30) <= t < time(12, 0) or time(13, 0) <= t < time(16, 0):
               return MarketSessionStatus.REGULAR
           elif time(9, 0) <= t < time(9, 30) or time(12, 0) <= t < time(13, 0):
               return MarketSessionStatus.PRE_MARKET
           elif time(16, 0) <= t < time(17, 0):
               return MarketSessionStatus.POST_MARKET
           return MarketSessionStatus.CLOSED
       
       return MarketSessionStatus.CLOSED
   ```

2. **Update is_fresh() method signature and implementation**:
   ```python
   def is_fresh(
       self,
       last_update: datetime,
       timeframe: Timeframe,
       now: Optional[datetime] = None,
       session_context: Optional[str] = None,
   ) -> bool:
       """Check if data is fresh.
       
       Args:
           last_update: Last update timestamp (UTC or timezone-aware)
           timeframe: Data timeframe
           now: Current time (defaults to now)
           session_context: Optional market code (e.g., 'cn_stock', 'us_futures')
                           When provided, staleness is suppressed during market-closed hours
                           if data was updated during the last trading session.
       
       Returns:
           True if data is fresh
       """
       if now is None:
           now = datetime.now(timezone.utc)
       
       # Ensure timezone-aware UTC
       if now.tzinfo is None:
           now = now.replace(tzinfo=timezone.utc)
       if last_update.tzinfo is None:
           last_update = last_update.replace(tzinfo=timezone.utc)
       
       threshold = self.thresholds.get(timeframe, 24 * 3600)
       age = (now - last_update).total_seconds()
       
       # Fast path: within threshold
       if age < threshold:
           return True
       
       # Session-aware staleness suppression
       if session_context:
           from .trading_sessions import get_session_status, MarketSessionStatus
           
           status = get_session_status(session_context, now)
           
           # If market is closed, check if data was updated in last session
           if status == MarketSessionStatus.CLOSED:
               from .trading_sessions import MARKET_TZ
               
               tz_name = MARKET_TZ.get(session_context.lower())
               if tz_name:
                   # Get market timezone
                   from zoneinfo import ZoneInfo
                   market_tz = ZoneInfo(tz_name)
                   
                   # Get dates in market timezone
                   last_update_market = last_update.astimezone(market_tz)
                   now_market = now.astimezone(market_tz)
                   
                   # If data was updated today in market timezone, not stale
                   if last_update_market.date() >= now_market.date():
                       return True
           
           # Even during active hours, allow some grace period beyond threshold
           # based on session status
           if status in (MarketSessionStatus.PRE_MARKET, MarketSessionStatus.POST_MARKET):
               # Give extra time during pre/post market
               extended_threshold = threshold * 1.5
               if age < extended_threshold:
                   return True
       
       return False
   ```

   ```python
   def get_freshness_status(
       self,
       last_update: datetime,
       timeframe: Timeframe,
       now: Optional[datetime] = None,
       session_context: Optional[str] = None,
   ) -> str:
       """Get freshness status.
       
       Returns:
           "fresh" / "stale" / "very_stale" / "session_closed"
       """
       if now is None:
           now = datetime.now(timezone.utc)
       
       if now.tzinfo is None:
           now = now.replace(tzinfo=timezone.utc)
       if last_update.tzinfo is None:
           last_update = last_update.replace(tzinfo=timezone.utc)
       
       threshold = self.thresholds.get(timeframe, 24 * 3600)
       age = (now - last_update).total_seconds()
       
       if session_context:
           from .trading_sessions import get_session_status, MARKET_TZ
           
           status = get_session_status(session_context, now)
           if status == MarketSessionStatus.CLOSED:
               tz_name = MARKET_TZ.get(session_context.lower())
               if tz_name:
                   from zoneinfo import ZoneInfo
                   market_tz = ZoneInfo(tz_name)
                   last_update_market = last_update.astimezone(market_tz)
                   now_market = now.astimezone(market_tz)
                   if last_update_market.date() >= now_market.date():
                       return "session_closed"
       
       if age < threshold:
           return "fresh"
       elif age < threshold * 2:
           return "stale"
       else:
           return "very_stale"
   ```
