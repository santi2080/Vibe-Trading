       ```python
       def per_source_stats(signals_per_source: Dict[str, List[dict]]) -> Dict[str, Dict]:
           """Per-strategy-source signal statistics (METR-03).

           Args:
               signals_per_source: {source_name: [TrendResult dicts]}

           Returns:
               {source_name: {count, avg_score, bullish_pct, readiness_ready_pct}}
           """
           result = {}
           for source, signals in signals_per_source.items():
               if not signals:
                   continue
               dirs = [s.get('direction', 'NEUTRAL') for s in signals]
               scores = [s.get('signed_score', 0) for s in signals]
               readies = [s.get('readiness', 'UNKNOWN') for s in signals]
               result[source] = {
                   'signal_count': len(signals),
                   'avg_score': round(sum(scores) / len(scores), 2),
                   'bullish_pct': round(dirs.count('BULL') / len(dirs), 4),
                   'bearish_pct': round(dirs.count('BEAR') / len(dirs), 4),
                   'neutral_pct': round(dirs.count('NEUTRAL') / len(dirs), 4),
                   'readiness_ready_pct': round(readies.count('READY') / len(readies), 4),
               }
           return result
       ```

    2. Add `equity_gap_check(equity_series: pd.Series) -> Dict[str, Any]` (RPT-03):
       ```python
       def equity_gap_check(equity_series: pd.Series) -> Dict[str, Any]:
           """Detect data quality issues from equity curve gaps (RPT-03)."""
           if len(equity_series) < 2:
               return {'gaps': 0, 'max_gap_days': 0, 'quality': 'insufficient_data'}
           rets = equity_series.pct_change().dropna()
           zero_days = int((rets == 0).sum())
           total_days = len(rets)
           quality = 'good' if zero_days / total_days < 0.1 else 'degraded'
           return {
               'total_days': total_days,
               'zero_return_days': zero_days,
               'gap_ratio': round(zero_days / total_days, 4),
               'quality': quality,
           }
       ```
