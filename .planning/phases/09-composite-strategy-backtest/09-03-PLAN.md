    ```python
    def make_engine(config: dict):
        \"\"\"Factory: build CompositeBacktestSignalEngine from config dict.\"\"\"
        # MTES v3 source
        mtes_cfg = MTESv3Config(
            fast_ema=config.get("mtes_fast_ema", 12),
            slow_ema=config.get("mtes_slow_ema", 26),
            adx_period=config.get("mtes_adx_period", 14),
            adx_threshold=float(config.get("mtes_adx_threshold", 25.0)),
            min_bull_ema_gap=float(config.get("mtes_min_bull_gap", 0.5)),
            min_bear_ema_gap=float(config.get("mtes_min_bear_gap", -0.5)),
        )
        mtes_strategy = MTESv3TrendStrategy(
            mtes_config=mtes_cfg,
            strategy_config=TrendStrategyConfig(),
        )

        # Enhanced SuperTrend source
        st_strategy = EnhancedSuperTrendStrategy(
            config=None,  # use EnhancedSuperTrendStrategy defaults
            strategy_config=TrendStrategyConfig(),
        )

        # Composite
        composite = CompositeTrendStrategy(
            sources=[mtes_strategy, st_strategy],
        )

        return CompositeBacktestSignalEngine(
            composite=composite,
            atr_multiplier=float(config.get("atr_multiplier", 2.0)),
            atr_period=int(config.get("atr_period", 14)),
        )


    class SignalEngine:
        \"\"\"Entry point called by runner.py.

        runner.py calls engine_cls() with no args (line 452).
        Then calls signal_engine.generate(data_map) for each bar.
        \"\"\"

        def __init__(self, config: dict | None = None):
            self._config = config or {}
            self._engine = make_engine(self._config)
            self._signals_output = None

        def generate(self, data_map: Dict[str, pd.DataFrame]) -> Dict[str, pd.Series]:
            \"\"\"Generate composite signals for backtest.\"\"\"
            return self._engine.generate(data_map)

        def get_signal_output(self):
            \"\"\"Return full signal output (key nodes + per-source) for artifact writing.\"\"\"
            return self._engine._recorder.to_output(
                self._engine._per_source_signals if hasattr(self._engine, '_per_source_signals') else {}
            )
    ```

    - `__init__` accepts optional `config` dict (passed by runner via engine_cls())
    - No import-time execution (all AST validation requirements from runner.py)
    - Imports are all safe (no top-level function calls)
