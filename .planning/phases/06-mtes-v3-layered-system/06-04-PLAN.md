```python
# agent/src/analysis/mtes_v3/mtes_v3.py

import pandas as pd
from typing import Optional
from dataclasses import dataclass

from .base import MTESv3Result, TrendBias, StrengthRating, EntrySignal
from .preprocessor import Preprocessor, PreprocessorConfig
from .layer1 import Layer1Integrator
from .layer2 import ADXStrengthFilter, MomentumDivergenceDetector
from .layer3 import EntryTiming

@dataclass
class MTESv3Config:
    """MTES v3 配置"""
    # 预过滤
    adx_prefilter_threshold: float = 20.0
    
    # Layer 2
    adx_strong_threshold: float = 30.0
    adx_ready_threshold: float = 25.0
    
    # Layer 3
    rsi_oversold: float = 35.0
    rsi_overbought: float = 65.0

class MTESv3:
    """MTES v3 分层递进趋势系统"""
    
    def __init__(self, config: Optional[MTESv3Config] = None):
        self.config = config or MTESv3Config()
        
        # 初始化各层
        self.preprocessor = Preprocessor(PreprocessorConfig(
            adx_threshold=self.config.adx_prefilter_threshold
        ))
        self.layer1 = Layer1Integrator()
        self.strength_filter = ADXStrengthFilter(
            adx_strong=self.config.adx_strong_threshold,
            adx_weak=self.config.adx_ready_threshold
        )
        self.divergence_detector = MomentumDivergenceDetector()
        self.entry_timing = EntryTiming(
            rsi_oversold=self.config.rsi_oversold,
            rsi_overbought=self.config.rsi_overbought
        )
    
    def analyze(
        self,
        df: pd.DataFrame,
        higher_timeframe: Optional[pd.DataFrame] = None
    ) -> MTESv3Result:
        """完整 MTES v3 分析"""
        
        # ===== Layer 0: 预处理 =====
        prefilter_result = self.preprocessor.analyze(df)
        
        if not prefilter_result.get("passed", False):
            return self._create_insufficient_result(prefilter_result)
        
        # ===== Layer 1: 大周期趋势锁定 =====
        mtf_trend = self.layer1.analyze(df)
        
        # ===== Layer 2: 趋势强度确认 =====
        strength = self.strength_filter.filter(df, mtf_trend.direction)
        
        # 动量背离检测
        divergence = self.divergence_detector.analyze(df)
        
        # ===== Layer 3: 入场时机 =====
        entry = self.entry_timing.find_entry(
            df,
            mtf_trend.direction,
            strength.rating
        )
        
        # ===== 综合评分 =====
        final_score = self._calculate_final_score(
            mtf_trend,
            strength,
            divergence,
            entry
        )
        
        return MTESv3Result(
            passed_prefilter=True,
            mtf_trend=mtf_trend,
            strength=strength,
            entry=entry,
            final_score=final_score,
            final_confidence=self._calculate_confidence(
                mtf_trend,
                strength,
                divergence
            )
        )
    
    def _calculate_final_score(
        self,
        mtf_trend: TrendBias,
        strength: StrengthRating,
        divergence,
        entry: EntrySignal
    ) -> float:
        """计算最终评分"""
        # 方向基础分
        if mtf_trend.direction == "BULL":
            base_score = 50 * mtf_trend.confidence
        elif mtf_trend.direction == "BEAR":
            base_score = -50 * mtf_trend.confidence
        else:
            base_score = 0
        
        # 强度加成
        strength_bonus = 0
        if strength.rating == "STRONG":
            strength_bonus = 30
        elif strength.rating == "READY":
            strength_bonus = 15
        elif strength.rating == "WEAK":
            strength_bonus = 0
        else:  # EXHAUSTED
            strength_bonus = -20
        
        # 背离惩罚
        divergence_penalty = -30 if divergence.detected else 0
        
        # 入场信号加成
        entry_bonus = 0
        if entry.signal == "LONG":
            entry_bonus = 20
        elif entry.signal == "SHORT":
            entry_bonus = -20
        
        final = base_score + strength_bonus + divergence_penalty + entry_bonus
        
        return max(-100, min(100, final))
    
    def _calculate_confidence(
        self,
        mtf_trend: TrendBias,
        strength: StrengthRating,
        divergence
    ) -> float:
        """计算综合置信度"""
        # Layer 1 置信度
        layer1_conf = mtf_trend.confidence * 0.4
        
        # Layer 2 置信度
        if strength.rating == "STRONG":
            layer2_conf = 0.3
        elif strength.rating == "READY":
            layer2_conf = 0.2
        else:
            layer2_conf = 0.1
        
        # 背离惩罚
        divergence_penalty = 0.1 if divergence.detected else 0
        
        return max(0, min(1, layer1_conf + layer2_conf - divergence_penalty))
    
    def _create_insufficient_result(self, prefilter_result) -> MTESv3Result:
        """创建预过滤失败结果"""
        return MTESv3Result(
            passed_prefilter=False,
            mtf_trend=TrendBias(direction="NEUTRAL", confidence=0, signals={}),
            strength=StrengthRating(rating="EXHAUSTED", adx_value=0, divergence=False, mtf_bias=None),
            entry=EntrySignal(signal="WAIT"),
            final_score=0,
            final_confidence=0
        )
```

```python
# agent/src/analysis/mtes_v3/adapter.py

from .mtes_v3 import MTESv3, MTESv3Result
from .base import TrendBias, StrengthRating, EntrySignal

class MTESv2Adapter:
    """MTES v2 格式适配器"""
    
    def __init__(self, v3: MTESv3):
        self.v3 = v3
    
    def analyze_v2_format(self, df, **kwargs) -> dict:
        """转换为 MTES v2 格式"""
        v3_result = self.v3.analyze(df, **kwargs)
        return self.to_v2_format(v3_result)
    
    def to_v2_format(self, v3_result: MTESv3Result) -> dict:
        """转换为 MTES v2 字典格式"""
        
        # 方向映射
        direction_map = {
            "BULL": "UP",
            "BEAR": "DOWN",
            "NEUTRAL": "-"
        }
        
        # 强度映射
        strength_map = {
            "STRONG": "STRONG",
            "READY": "CONFIRMED",
            "WEAK": "WEAK",
            "EXHAUSTED": "NEUTRAL"
        }
        
        # 趋势状态映射
        if v3_result.final_score >= 60:
            trend_state = "BULL_STRONG"
        elif v3_result.final_score >= 30:
            trend_state = "BULL_CONFIRMED"
        elif v3_result.final_score >= 0:
            trend_state = "BULL_EARLY"
        elif v3_result.final_score >= -30:
            trend_state = "BEAR_EARLY"
        elif v3_result.final_score >= -60:
            trend_state = "BEAR_CONFIRMED"
        else:
            trend_state = "BEAR_STRONG"
        
        return {
            # MTES v2 兼容字段
            "trend_score": v3_result.final_score,
            "direction": direction_map.get(v3_result.mtf_trend.direction, "-"),
            "confidence": v3_result.final_confidence,
            "trend_state": trend_state,
            "regime": "TRENDING" if v3_result.strength.rating != "EXHAUSTED" else "CHOPPY",
            
            # MTES v3 扩展字段
            "v3": {
                "layer1": {
                    "smc_trend": v3_result.mtf_trend.direction,
                    "elder_trend": v3_result.mtf_trend.signals.get("elder", {}).get("layer1_trend"),
                    "ichimoku_trend": v3_result.mtf_trend.signals.get("ichimoku", {}).get("trend"),
                    "confidence": v3_result.mtf_trend.confidence
                },
                "layer2": {
                    "adx_value": v3_result.strength.adx_value,
                    "rating": v3_result.strength.rating,
                    "divergence": v3_result.strength.divergence
                },
                "layer3": {
                    "entry_signal": v3_result.entry.signal,
                    "entry_price": v3_result.entry.entry_price,
                    "stop_loss": v3_result.entry.stop_loss,
                    "reason": v3_result.entry.reason
                }
            }
        }
    
    def analyze_and_format(self, df, **kwargs) -> dict:
        """分析并返回 v2 格式"""
        return self.analyze_v2_format(df, **kwargs)
```

```python
# tests/test_mtes_v3/test_mtes_v3_integration.py

import pytest
import pandas as pd
import numpy as np
from agent.src.analysis.mtes_v3 import MTESv3, MTESv2Adapter

@pytest.fixture
def sample_trending_data():
    """趋势数据"""
    np.random.seed(42)
    n = 200
    dates = pd.date_range('2025-01-01', periods=n, freq='D')
    
    # 上涨趋势
    trend = np.linspace(0, 50, n)
    noise = np.random.randn(n) * 2
    
    df = pd.DataFrame({
        'open': 100 + trend + noise,
        'high': 100 + trend + noise + abs(np.random.randn(n) * 2),
        'low': 100 + trend + noise - abs(np.random.randn(n) * 2),
        'close': 100 + trend + noise,
        'volume': np.random.randint(1000, 5000, n)
    }, index=dates)
    return df

def test_mtes_v3_full_analysis(sample_trending_data):
    """完整 MTES v3 分析"""
    mtes = MTESv3()
    result = mtes.analyze(sample_trending_data)
    
    assert result.passed_prefilter == True
    assert result.final_score > 0  # 应该是多头
    assert result.final_confidence > 0
    assert result.mtf_trend.direction in ["BULL", "NEUTRAL"]

def test_mtes_v2_adapter():
    """MTES v2 适配器"""
    mtes = MTESv3()
    adapter = MTESv2Adapter(mtes)
    
    result = adapter.analyze_v2_format(sample_trending_data())
    
    # 检查 v2 格式
    assert "trend_score" in result
    assert "direction" in result
    assert "confidence" in result
    assert "trend_state" in result
    
    # 检查 v3 扩展
    assert "v3" in result
    assert "layer1" in result["v3"]
    assert "layer2" in result["v3"]
    assert "layer3" in result["v3"]

def test_batch_analysis():
    """批量分析"""
    mtes = MTESv3()
    
    symbols = ["GC=F", "SI=F", "CL=F"]
    data_dict = {}
    
    for sym in symbols:
        data_dict[sym] = sample_trending_data()
    
    results = {}
    for sym, df in data_dict.items():
        results[sym] = mtes.analyze(df)
    
    assert len(results) == 3
    for sym, result in results.items():
        assert result.passed_prefilter == True

def test_performance():
    """性能测试"""
    import time
    
    mtes = MTESv3()
    df = sample_trending_data()
    
    start = time.time()
    for _ in range(100):
        mtes.analyze(df)
    elapsed = time.time() - start
    
    # 100 次分析应在 10 秒内完成
    assert elapsed < 10, f"Too slow: {elapsed:.2f}s for 100 analyses"
```
