# agent/src/analysis/mtes_v3/base.py

from abc import ABC, abstractmethod
from dataclasses import dataclass
from typing import Literal, Optional
import pandas as pd

@dataclass
class TrendBias:
    """趋势偏见"""
    direction: Literal["BULL", "BEAR", "NEUTRAL"]
    confidence: float  # 0-1
    signals: dict     # 各子信号详情

@dataclass
class StrengthRating:
    """趋势强度"""
    rating: Literal["STRONG", "READY", "WEAK", "EXHAUSTED"]
    adx_value: float
    divergence: bool
    regime: str

@dataclass
class EntrySignal:
    """入场信号"""
    signal: Literal["LONG", "SHORT", "WAIT"]
    entry_price: Optional[float] = None
    stop_loss: Optional[float] = None
    reason: Optional[str] = None

@dataclass
class MTESv3Result:
    """MTES v3 最终结果"""
    passed_prefilter: bool
    mtf_trend: TrendBias
    strength: StrengthRating
    entry: EntrySignal
    final_score: float  # -100 ~ +100
    final_confidence: float

class BaseLayer(ABC):
    """层级抽象基类"""
    
    @abstractmethod
    def analyze(self, df: pd.DataFrame, **kwargs) -> any:
        """分析入口"""
        pass
    
    @abstractmethod
    def validate(self, df: pd.DataFrame) -> bool:
        """数据验证"""
        pass
```

# agent/src/analysis/mtes_v3/preprocessor.py

import pandas as pd
import numpy as np
from typing import Optional
from .base import BaseLayer
from src.indicators.trend import ADXIndicator

@dataclass
class PreprocessorConfig:
    adx_threshold: float = 20.0
    min_data_points: int = 200
    min_volume: float = 0.0
    price_col: str = 'close'

class Preprocessor(BaseLayer):
    """Layer 0: 预处理层"""
    
    def __init__(self, config: Optional[PreprocessorConfig] = None):
        self.config = config or PreprocessorConfig()
        self.adx_indicator = ADXIndicator()
    
    def validate(self, df: pd.DataFrame) -> bool:
        """数据完整性验证"""
        required_cols = ['open', 'high', 'low', 'close']
        return all(col in df.columns for col in required_cols)
    
    def analyze(self, df: pd.DataFrame, **kwargs) -> dict:
        """预处理分析"""
        if not self.validate(df):
            return {"passed": False, "reason": "invalid_data"}
        
        # 1. 数据量检查
        if len(df) < self.config.min_data_points:
            return {"passed": False, "reason": "insufficient_data"}
        
        # 2. ADX 预过滤
        adx_value = self.adx_indicator.calculate(df)
        if adx_value < self.config.adx_threshold:
            return {
                "passed": False, 
                "reason": "adx_below_threshold",
                "adx_value": adx_value
            }
        
        # 3. 成交量检查
        if self.config.min_volume > 0:
            avg_volume = df['volume'].mean() if 'volume' in df.columns else 0
            if avg_volume < self.config.min_volume:
                return {"passed": False, "reason": "low_volume"}
        
        return {
            "passed": True,
            "adx_value": adx_value,
            "data_points": len(df)
        }
    
    def filter_batch(self, data_dict: dict[str, pd.DataFrame]) -> dict[str, bool]:
        """批量预过滤"""
        results = {}
        for symbol, df in data_dict.items():
            results[symbol] = self.analyze(df)["passed"]
        return results
```

# agent/src/analysis/mtes_v3/layer1/smc_analyzer.py

import pandas as pd
import numpy as np
from typing import Literal, Optional
from dataclasses import dataclass
from ..base import BaseLayer

@dataclass
class Swing:
    """波段点"""
    index: int
    timestamp: pd.Timestamp
    price: float
    swing_type: Literal["HH", "HL", "LH", "LL"]
    is_bullish: bool  # True = 多头控制

@dataclass
class MarketStructureResult:
    """市场结构结果"""
    trend: Literal["BULL", "BEAR", "NEUTRAL"]
    confidence: float
    swings: list[Swing]
    bos_confirmed: bool
    mss_confirmed: bool
    last_swing_high: Optional[float]
    last_swing_low: Optional[float]
    liquidity_sweeps: list[dict]

class SwingDetector:
    """波段高低点检测"""
    
    def __init__(self, threshold: float = 0.005):
        self.threshold = threshold  # 波动阈值 0.5%
    
    def detect(self, df: pd.DataFrame) -> list[Swing]:
        """检测所有波段高低点"""
        swings = []
        
        highs = df['high'].values
        lows = df['low'].values
        closes = df['close'].values
        timestamps = df.index
        
        for i in range(2, len(df) - 2):
            # 检测波段高点 (HH 或 LH)
            if highs[i] > highs[i-1] and highs[i] > highs[i-2] and \
               highs[i] > highs[i+1] and highs[i] > highs[i+2]:
                swing_type = self._classify_high(i, swings, highs)
                swings.append(Swing(
                    index=i,
                    timestamp=timestamps[i] if hasattr(timestamps[i], '__iter__') else pd.Timestamp(timestamps[i]),
                    price=highs[i],
                    swing_type=swing_type,
                    is_bullish=False
                ))
            
            # 检测波段低点 (HL 或 LL)
            if lows[i] < lows[i-1] and lows[i] < lows[i-2] and \
               lows[i] < lows[i+1] and lows[i] < lows[i+2]:
                swing_type = self._classify_low(i, swings, lows)
                swings.append(Swing(
                    index=i,
                    timestamp=timestamps[i] if hasattr(timestamps[i], '__iter__') else pd.Timestamp(timestamps[i]),
                    price=lows[i],
                    swing_type=swing_type,
                    is_bullish=True
                ))
        
        return sorted(swings, key=lambda s: s.index)
    
    def _classify_high(self, idx: int, prev_swings: list[Swing], highs: np.ndarray) -> str:
        """分类波段高点"""
        prev_highs = [s.price for s in prev_swings if s.swing_type in ["HH", "LH"]]
        if prev_highs and highs[idx] > max(prev_highs):
            return "HH"  # 更高的波段高点
        return "LH"
    
    def _classify_low(self, idx: int, prev_swings: list[Swing], lows: np.ndarray) -> str:
        """分类波段低点"""
        prev_lows = [s.price for s in prev_swings if s.swing_type in ["HL", "LL"]]
        if prev_lows and lows[idx] < min(prev_lows):
            return "LL"  # 更低的波段低点
        return "HL"

class SMCAnalyzer(BaseLayer):
    """SMC 市场结构分析"""
    
    def __init__(self, swing_threshold: float = 0.005):
        self.swing_detector = SwingDetector(swing_threshold)
    
    def validate(self, df: pd.DataFrame) -> bool:
        return len(df) >= 50
    
    def analyze(self, df: pd.DataFrame, **kwargs) -> MarketStructureResult:
        """完整 SMC 分析"""
        # 1. 检测波段
        swings = self.swing_detector.detect(df)
        
        if len(swings) < 4:
            return MarketStructureResult(
                trend="NEUTRAL",
                confidence=0.0,
                swings=swings,
                bos_confirmed=False,
                mss_confirmed=False,
                last_swing_high=None,
                last_swing_low=None,
                liquidity_sweeps=[]
            )
        
        # 2. 确定当前趋势
        trend, confidence = self._determine_trend(swings)
        
        # 3. 检测 BOS
        bos_confirmed = self._detect_bos(swings, trend)
        
        # 4. 检测 MSS
        mss_confirmed = self._detect_mss(swings, trend)
        
        # 5. 检测流动性扫掠
        liquidity_sweeps = self._detect_sweeps(df, swings)
        
        # 6. 获取最近高低点
        recent_highs = [s for s in swings if s.swing_type in ["HH", "LH"]]
        recent_lows = [s for s in swings if s.swing_type in ["HL", "LL"]]
        last_swing_high = recent_highs[-1].price if recent_highs else None
        last_swing_low = recent_lows[-1].price if recent_lows else None
        
        return MarketStructureResult(
            trend=trend,
            confidence=confidence,
            swings=swings,
            bos_confirmed=bos_confirmed,
            mss_confirmed=mss_confirmed,
            last_swing_high=last_swing_high,
            last_swing_low=last_swing_low,
            liquidity_sweeps=liquidity_sweeps
        )
    
    def _determine_trend(self, swings: list[Swing]) -> tuple[str, float]:
        """确定趋势方向"""
        highs = [s for s in swings if s.swing_type in ["HH", "LH"]]
        lows = [s for s in swings if s.swing_type in ["HL", "LL"]]
        
        if len(highs) < 2 or len(lows) < 2:
            return "NEUTRAL", 0.0
        
        # 上升趋势: HH 和 HL 都在抬高
        hh_trend = highs[-1].price > highs[-2].price if len(highs) >= 2 else False
        hl_trend = lows[-1].price > lows[-2].price if len(lows) >= 2 else False
        
        if hh_trend and hl_trend:
            return "BULL", 0.8
        
        # 下降趋势: LH 和 LL 都在降低
        lh_trend = highs[-1].price < highs[-2].price if len(highs) >= 2 else False
        ll_trend = lows[-1].price < lows[-2].price if len(lows) >= 2 else False
        
        if lh_trend and ll_trend:
            return "BEAR", 0.8
        
        return "NEUTRAL", 0.3
    
    def _detect_bos(self, swings: list[Swing], trend: str) -> bool:
        """检测结构破坏 (Break of Structure)"""
        if trend == "NEUTRAL":
            return False
        
        highs = [s for s in swings if s.swing_type in ["HH", "LH"]]
        lows = [s for s in swings if s.swing_type in ["HL", "LL"]]
        
        if trend == "BULL" and len(highs) >= 2:
            # 上升趋势中突破前高
            return highs[-1].price > highs[-2].price
        elif trend == "BEAR" and len(lows) >= 2:
            # 下降趋势中跌破前低
            return lows[-1].price < lows[-2].price
        
        return False
    
    def _detect_mss(self, swings: list[Swing], trend: str) -> bool:
        """检测市场结构转变 (Market Structure Shift)"""
        if len(swings) < 4:
            return False
        
        highs = [s for s in swings if s.swing_type in ["HH", "LH"]]
        lows = [s for s in swings if s.swing_type in ["HL", "LL"]]
        
        if trend == "BULL" and len(highs) >= 2:
            # 多头趋势中，价格跌破前低 → MSS
            return lows[-1].price < lows[-2].price
        elif trend == "BEAR" and len(highs) >= 2:
            # 空头趋势中，价格突破前高 → MSS
            return highs[-1].price > highs[-2].price
        
        return False
