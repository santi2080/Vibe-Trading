```python
# agent/src/analysis/mtes_v3/layer1/elder_screen.py

import pandas as pd
import numpy as np
from typing import Literal, Optional
from dataclasses import dataclass
from ..base import BaseLayer

@dataclass
class ElderSignal:
    """Elder 三重滤网信号"""
    layer1_trend: Literal["BULL", "BEAR", "NEUTRAL"]
    layer2_pullback: bool
    layer3_trigger: Literal["READY", "WAIT"]
    macd_histogram_slope: float
    rsi_value: float
    macd_histogram_values: list

class ElderTripleScreen(BaseLayer):
    """Elder 三重滤网"""
    
    def __init__(
        self,
        macd_fast: int = 12,
        macd_slow: int = 26,
        macd_signal: int = 9,
        rsi_period: int = 14,
        rsi_oversold: float = 30,
        rsi_overbought: float = 70
    ):
        self.macd_fast = macd_fast
        self.macd_slow = macd_slow
        self.macd_signal = macd_signal
        self.rsi_period = rsi_period
        self.rsi_oversold = rsi_oversold
        self.rsi_overbought = rsi_overbought
    
    def validate(self, df: pd.DataFrame) -> bool:
        return len(df) >= max(self.macd_slow, self.rsi_period) + 2
    
    def calculate_macd(self, df: pd.DataFrame) -> tuple:
        """计算 MACD"""
        close = df['close']
        ema_fast = close.ewm(span=self.macd_fast, adjust=False).mean()
        ema_slow = close.ewm(span=self.macd_slow, adjust=False).mean()
        macd_line = ema_fast - ema_slow
        signal_line = macd_line.ewm(span=self.macd_signal, adjust=False).mean()
        histogram = macd_line - signal_line
        return macd_line, signal_line, histogram
    
    def calculate_macd_histogram_slope(self, histogram: pd.Series, lookback: int = 2) -> float:
        """计算 MACD 柱状图斜率"""
        if len(histogram) < lookback + 1:
            return 0.0
        recent = histogram.iloc[-lookback:]
        x = np.arange(len(recent))
        slope = np.polyfit(x, recent.values, 1)[0]
        return slope
    
    def calculate_rsi(self, df: pd.DataFrame) -> float:
        """计算 RSI"""
        close = df['close']
        delta = close.diff()
        gain = delta.where(delta > 0, 0).rolling(self.rsi_period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(self.rsi_period).mean()
        rs = gain / loss
        rsi = 100 - (100 / (1 + rs))
        return rsi.iloc[-1]
    
    def layer1_mtf_trend(self, df: pd.DataFrame) -> str:
        """第一滤网: MACD 柱状图斜率"""
        _, _, histogram = self.calculate_macd(df)
        slope = self.calculate_macd_histogram_slope(histogram)
        
        if slope > 0:
            return "BULL"
        elif slope < 0:
            return "BEAR"
        return "NEUTRAL"
    
    def layer2_pullback_extremity(self, df: pd.DataFrame, mtf_trend: str) -> bool:
        """第二滤网: RSI 超卖检测"""
        rsi = self.calculate_rsi(df)
        
        if mtf_trend == "BULL" and rsi < self.rsi_oversold:
            return True
        elif mtf_trend == "BEAR" and rsi > self.rsi_overbought:
            return True
        return False
    
    def layer3_trigger(self, df: pd.DataFrame, pullback_low: float = None) -> str:
        """第三滤网: Buy Stop 突破执行"""
        if pullback_low is None:
            pullback_low = df['low'].iloc[-2] if len(df) >= 2 else df['low'].iloc[-1]
        
        current_high = df['high'].iloc[-1]
        trigger_price = pullback_low + 0.001  # 1 tick above
        
        if current_high > trigger_price:
            return "READY"
        return "WAIT"
    
    def analyze(self, df: pd.DataFrame, **kwargs) -> ElderSignal:
        """完整三重滤网分析"""
        # Layer 1
        layer1_trend = self.layer1_mtf_trend(df)
        
        # Layer 2
        layer2_pullback = self.layer2_pullback_extremity(df, layer1_trend)
        
        # Layer 3
        layer3_trigger = self.layer3_trigger(df)
        
        # 计算指标值
        _, _, histogram = self.calculate_macd(df)
        macd_histogram_slope = self.calculate_macd_histogram_slope(histogram)
        rsi_value = self.calculate_rsi(df)
        
        return ElderSignal(
            layer1_trend=layer1_trend,
            layer2_pullback=layer2_pullback,
            layer3_trigger=layer3_trigger,
            macd_histogram_slope=macd_histogram_slope,
            rsi_value=rsi_value,
            macd_histogram_values=histogram.tail(5).tolist()
        )
```

```python
# agent/src/analysis/mtes_v3/layer1/ichimoku.py

import pandas as pd
import numpy as np
from typing import Literal, Optional
from dataclasses import dataclass
from ..base import BaseLayer

@dataclass
class IchimokuSignal:
    """Ichimoku 信号"""
    trend: Literal["BULL", "BEAR", "NEUTRAL"]
    price_above_cloud: bool
    tenkan_above_kijun: bool
    cloud_bullish: bool  # Span A > Span B
    chikou_above_price: bool  # 延迟线确认
    cloud_thickness: float  # 云带厚度
    confidence: float

class IchimokuCloud(BaseLayer):
    """一目均衡表"""
    
    def __init__(
        self,
        tenkan_period: int = 9,
        kijun_period: int = 26,
        senkou_b_period: int = 52,
        displacement: int = 26
    ):
        self.tenkan_period = tenkan_period
        self.kijun_period = kijun_period
        self.senkou_b_period = senkou_b_period
        self.displacement = displacement
    
    def validate(self, df: pd.DataFrame) -> bool:
        return len(df) >= self.senkou_b_period + self.displacement + 2
    
    def calculate_tenkan(self, df: pd.DataFrame) -> pd.Series:
        """转换线: (最高价 + 最低价) / 2"""
        high = df['high'].rolling(self.tenkan_period).max()
        low = df['low'].rolling(self.tenkan_period).min()
        return (high + low) / 2
    
    def calculate_kijun(self, df: pd.DataFrame) -> pd.Series:
        """基准线: (26期最高价 + 26期最低价) / 2"""
        high = df['high'].rolling(self.kijun_period).max()
        low = df['low'].rolling(self.kijun_period).min()
        return (high + low) / 2
    
    def calculate_senkou_a(self, df: pd.DataFrame) -> pd.Series:
        """先行跨带 A"""
        tenkan = self.calculate_tenkan(df)
        kijun = self.calculate_kijun(df)
        senkou_a = (tenkan + kijun) / 2
        return senkou_a.shift(self.displacement)  # 向前位移
    
    def calculate_senkou_b(self, df: pd.DataFrame) -> pd.Series:
        """先行跨带 B"""
        high = df['high'].rolling(self.senkou_b_period).max()
        low = df['low'].rolling(self.senkou_b_period).min()
        senkou_b = (high + low) / 2
        return senkou_b.shift(self.displacement)  # 向前位移
    
    def calculate_chikou(self, df: pd.DataFrame) -> pd.Series:
        """延迟线: 当前收盘价向后移 26"""
        return df['close'].shift(-self.displacement)
    
    def analyze(self, df: pd.DataFrame, **kwargs) -> IchimokuSignal:
        """完整 Ichimoku 分析"""
        # 计算各线
        tenkan = self.calculate_tenkan(df)
        kijun = self.calculate_kijun(df)
        senkou_a = self.calculate_senkou_a(df)
        senkou_b = self.calculate_senkou_b(df)
        chikou = self.calculate_chikou(df)
        
        # 当前值
        current_close = df['close'].iloc[-1]
        current_tenkan = tenkan.iloc[-1]
        current_kijun = kijun.iloc[-1]
        current_senkou_a = senkou_a.iloc[-1]
        current_senkou_b = senkou_b.iloc[-1]
        current_chikou = chikou.iloc[-1]
        
        # 云带当前值
        cloud_top = max(current_senkou_a, current_senkou_b)
        cloud_bottom = min(current_senkou_a, current_senkou_b)
        cloud_thickness = cloud_top - cloud_bottom
        
        # 判定条件
        price_above_cloud = current_close > cloud_top
        price_in_cloud = cloud_bottom <= current_close <= cloud_top
        price_below_cloud = current_close < cloud_bottom
        
        tenkan_above_kijun = current_tenkan > current_kijun
        cloud_bullish = current_senkou_a > current_senkou_b
        chikou_above_price = current_chikou > current_close
        
        # 综合判定
        bullish_signals = sum([
            price_above_cloud,
            tenkan_above_kijun,
            cloud_bullish,
            chikou_above_price
        ])
        
        if bullish_signals >= 3:
            trend = "BULL"
            confidence = bullish_signals / 4.0
        elif bullish_signals <= 1:
            trend = "BEAR"
            confidence = (4 - bullish_signals) / 4.0
        else:
            trend = "NEUTRAL"
            confidence = 0.5
        
        return IchimokuSignal(
            trend=trend,
            price_above_cloud=price_above_cloud,
            tenkan_above_kijun=tenkan_above_kijun,
            cloud_bullish=cloud_bullish,
            chikou_above_price=chikou_above_price,
            cloud_thickness=cloud_thickness,
            confidence=confidence
        )
```

```python
# agent/src/analysis/mtes_v3/layer1/__init__.py

from .smc_analyzer import SMCAnalyzer, MarketStructureResult
from .elder_screen import ElderTripleScreen, ElderSignal
from .ichimoku import IchimokuCloud, IchimokuSignal
from ..base import TrendBias

class Layer1Integrator:
    """Layer 1 三重共振整合器"""
    
    def __init__(self):
        self.smc = SMCAnalyzer()
        self.elder = ElderTripleScreen()
        self.ichimoku = IchimokuCloud()
    
    def analyze(self, df: pd.DataFrame) -> TrendBias:
        """三层共振确认趋势"""
        # SMC
        smc_result = self.smc.analyze(df)
        
        # Elder
        elder_result = self.elder.analyze(df)
        
        # Ichimoku
        ichimoku_result = self.ichimoku.analyze(df)
        
        # 信号映射
        signals = [
            smc_result.trend,
            elder_result.layer1_trend,
            ichimoku_result.trend
        ]
        
        bull_count = signals.count("BULL")
        bear_count = signals.count("BEAR")
        
        # 共振判定
        if bull_count >= 2:
            direction = "BULL"
            confidence = bull_count / 3.0
        elif bear_count >= 2:
            direction = "BEAR"
            confidence = bear_count / 3.0
        else:
            direction = "NEUTRAL"
            confidence = 0.3
        
        return TrendBias(
            direction=direction,
            confidence=confidence,
            signals={
                "smc": smc_result.__dict__,
                "elder": elder_result.__dict__,
                "ichimoku": ichimoku_result.__dict__
            }
        )
```
