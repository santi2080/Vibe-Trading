import pandas as pd
import numpy as np
from typing import Literal
from dataclasses import dataclass
from ..base import BaseLayer, StrengthRating
from src.indicators.trend import ADXIndicator

class ADXStrengthFilter(BaseLayer):
    """ADX 趋势强度过滤器"""
    
    def __init__(
        self,
        adx_strong: float = 30.0,
        adx_weak: float = 25.0,
        adx_min: float = 20.0
    ):
        self.adx_strong = adx_strong      # STRONG 阈值
        self.adx_weak = adx_weak          # READY 阈值
        self.adx_min = adx_min            # 最低阈值
        self.adx_indicator = ADXIndicator()
    
    def validate(self, df: pd.DataFrame) -> bool:
        return len(df) >= 30
    
    def calculate_adx_slope(self, df: pd.DataFrame, lookback: int = 5) -> float:
        """计算 ADX 斜率"""
        adx_series = self.adx_indicator.calculate_series(df)
        if len(adx_series) < lookback:
            return 0.0
        
        recent = adx_series.tail(lookback).values
        x = np.arange(len(recent))
        slope = np.polyfit(x, recent, 1)[0]
        return slope
    
    def filter(self, df: pd.DataFrame, mtf_bias: str = None) -> StrengthRating:
        """趋势强度评级"""
        adx_value = self.adx_indicator.calculate(df)
        adx_slope = self.calculate_adx_slope(df)
        adx_rising = adx_slope > 0
        
        # 强度判定
        if adx_value >= self.adx_strong and adx_rising:
            rating = "STRONG"
        elif adx_value >= self.adx_weak:
            rating = "READY"
        elif adx_value >= self.adx_min:
            rating = "WEAK"
        else:
            rating = "EXHAUSTED"
        
        return StrengthRating(
            rating=rating,
            adx_value=adx_value,
            adx_slope=adx_slope,
            adx_rising=adx_rising,
            mtf_bias=mtf_bias
        )

# agent/src/analysis/mtes_v3/layer2/divergence.py

import pandas as pd
import numpy as np
from typing import Literal, Optional
from dataclasses import dataclass
from ..base import BaseLayer

@dataclass
class DivergenceResult:
    """背离检测结果"""
    detected: bool
    divergence_type: Optional[Literal["BULLISH", "BEARISH", "HIDDEN_BULL", "HIDDEN_BEAR"]]
    strength: float  # 0-1
    price_swing_high: Optional[float]
    price_swing_low: Optional[float]
    indicator_swing_high: Optional[float]
    indicator_swing_low: Optional[float]

class MomentumDivergenceDetector(BaseLayer):
    """动量背离检测"""
    
    def __init__(
        self,
        lookback: int = 50,
        min_swing_size: float = 0.02,  # 最小波动幅度 2%
        confirmation_bars: int = 3
    ):
        self.lookback = lookback
        self.min_swing_size = min_swing_size
        self.confirmation_bars = confirmation_bars
    
    def validate(self, df: pd.DataFrame) -> bool:
        return len(df) >= self.lookback
    
    def find_price_swing_highs(self, prices: np.ndarray) -> list:
        """寻找价格波段高点"""
        highs = []
        for i in range(2, len(prices) - 2):
            if (prices[i] > prices[i-1] and prices[i] > prices[i-2] and
                prices[i] > prices[i+1] and prices[i] > prices[i+2]):
                highs.append((i, prices[i]))
        return highs
    
    def find_price_swing_lows(self, prices: np.ndarray) -> list:
        """寻找价格波段低点"""
        lows = []
        for i in range(2, len(prices) - 2):
            if (prices[i] < prices[i-1] and prices[i] < prices[i-2] and
                prices[i] < prices[i+1] and prices[i] < prices[i+2]):
                lows.append((i, prices[i]))
        return lows
    
    def detect_divergence(
        self,
        prices: np.ndarray,
        indicator: np.ndarray
    ) -> DivergenceResult:
        """检测背离"""
        # 找到最近两个波段高/低点
        price_highs = self.find_price_swing_highs(prices)
        price_lows = self.find_price_swing_lows(prices)
        
        if len(price_highs) < 2 or len(price_lows) < 2:
            return DivergenceResult(
                detected=False,
                divergence_type=None,
                strength=0.0,
                price_swing_high=None,
                price_swing_low=None,
                indicator_swing_high=None,
                indicator_swing_low=None
            )
        
        # 检查顶背离 (价格创新高，指标未创新高)
        recent_highs = price_highs[-2:]
        if prices[-1] > recent_highs[0][1] > recent_highs[1][1]:  # 价格创新高
            # 检查指标是否背离
            indicator_at_highs = [
                indicator[h[0]] for h in recent_highs
                if h[0] < len(indicator)
            ]
            if len(indicator_at_highs) >= 2:
                if indicator[-1] < max(indicator_at_highs):  # 指标未创新高
                    return DivergenceResult(
                        detected=True,
                        divergence_type="BEARISH",
                        strength=self._calculate_strength(
                            prices[-1], recent_highs[-1][1],
                            indicator[-1], max(indicator_at_highs)
                        ),
                        price_swing_high=prices[-1],
                        price_swing_low=None,
                        indicator_swing_high=indicator[-1],
                        indicator_swing_low=None
                    )
        
        # 检查底背离 (价格创新低，指标未创新低)
        recent_lows = price_lows[-2:]
        if prices[-1] < recent_lows[0][1] < recent_lows[1][1]:  # 价格创新低
            indicator_at_lows = [
                indicator[l[0]] for l in recent_lows
                if l[0] < len(indicator)
            ]
            if len(indicator_at_lows) >= 2:
                if indicator[-1] > min(indicator_at_lows):  # 指标未创新低
                    return DivergenceResult(
                        detected=True,
                        divergence_type="BULLISH",
                        strength=self._calculate_strength(
                            min(indicator_at_lows), indicator[-1],
                            recent_lows[-1][1], prices[-1]
                        ),
                        price_swing_high=None,
                        price_swing_low=prices[-1],
                        indicator_swing_high=None,
                        indicator_swing_low=indicator[-1]
                    )
        
        return DivergenceResult(
            detected=False,
            divergence_type=None,
            strength=0.0,
            price_swing_high=None,
            price_swing_low=None,
            indicator_swing_high=None,
            indicator_swing_low=None
        )
    
    def _calculate_strength(self, price1, price2, ind1, ind2) -> float:
        """计算背离强度"""
        price_diff = abs(price1 - price2) / price1
        ind_diff = abs(ind1 - ind2) / abs(ind1) if ind1 != 0 else 0
        return min(1.0, price_diff / ind_diff) if ind_diff > 0 else 0.0
    
    def analyze(self, df: pd.DataFrame, **kwargs) -> DivergenceResult:
        """使用 MACD 检测背离"""
        # 计算 MACD
        close = df['close'].values
        ema_fast = pd.Series(close).ewm(span=12, adjust=False).mean().values
        ema_slow = pd.Series(close).ewm(span=26, adjust=False).mean().values
        macd = ema_fast - ema_slow
        
        return self.detect_divergence(close, macd)

# agent/src/analysis/mtes_v3/layer3/fvg_detector.py

import pandas as pd
import numpy as np
from typing import Literal, Optional
from dataclasses import dataclass
from ..base import BaseLayer

@dataclass
class FVG:
    """公允价值缺口"""
    index: int
    timestamp: pd.Timestamp
    upper_bound: float
    lower_bound: float
    size: float
    direction: Literal["BULLISH", "BEARISH"]
    filled: bool = False

class FVGDetector(BaseLayer):
    """公允价值缺口 (Fair Value Gap) 检测"""
    
    def __init__(self, min_gap_percent: float = 0.1):
        self.min_gap_percent = min_gap_percent  # 最小缺口百分比
    
    def validate(self, df: pd.DataFrame) -> bool:
        return len(df) >= 3
    
    def detect(self, df: pd.DataFrame) -> list[FVG]:
        """检测所有 FVG"""
        fvgs = []
        
        for i in range(2, len(df)):
            candle1 = df.iloc[i-2]
            candle2 = df.iloc[i-1]
            candle3 = df.iloc[i]
            
            # 计算中点
            mid1 = (candle1['high'] + candle1['low']) / 2
            mid2 = (candle2['high'] + candle2['low']) / 2
            
            # 检测看涨 FVG (第三根K线低点 > 第一根K线高点)
            if candle3['low'] > candle1['high']:
                gap_size = candle3['low'] - candle1['high']
                gap_percent = gap_size / candle1['high'] * 100
                
                if gap_percent >= self.min_gap_percent:
                    fvgs.append(FVG(
                        index=i,
                        timestamp=candle3.name if hasattr(candle3, 'name') else pd.Timestamp.now(),
                        upper_bound=candle3['low'],
                        lower_bound=candle1['high'],
                        size=gap_size,
                        direction="BULLISH"
                    ))
            
            # 检测看跌 FVG (第三根K线高点 < 第一根K线低点)
            if candle3['high'] < candle1['low']:
                gap_size = candle1['low'] - candle3['high']
                gap_percent = gap_size / candle1['low'] * 100
                
                if gap_percent >= self.min_gap_percent:
                    fvgs.append(FVG(
                        index=i,
                        timestamp=candle3.name if hasattr(candle3, 'name') else pd.Timestamp.now(),
                        upper_bound=candle1['low'],
                        lower_bound=candle3['high'],
                        size=gap_size,
                        direction="BEARISH"
                    ))
        
        return fvgs
    
    def find_unfilled_fvg(self, df: pd.DataFrame) -> list[FVG]:
        """寻找未填补的 FVG"""
        all_fvgs = self.detect(df)
        unfilled = []
        
        current_price = df['close'].iloc[-1]
        
        for fvg in all_fvgs:
            # 检查是否被填补
            if fvg.direction == "BULLISH":
                if current_price > fvg.lower_bound:
                    fvg.filled = True
                else:
                    unfilled.append(fvg)
            else:  # BEARISH
                if current_price < fvg.upper_bound:
                    fvg.filled = True
                else:
                    unfilled.append(fvg)
        
        return unfilled
    
    def is_price_near_fvg(self, df: pd.DataFrame, tolerance: float = 0.001) -> Optional[FVG]:
        """检测价格是否接近 FVG 区域"""
        unfilled = self.find_unfilled_fvg(df)
        
        if not unfilled:
            return None
        
        current_low = df['low'].iloc[-1]
        current_high = df['high'].iloc[-1]
        
        for fvg in unfilled[-3:]:  # 只检查最近3个
            # 看涨 FVG: 价格回落到上边界附近
            if fvg.direction == "BULLISH":
                if fvg.upper_bound * (1 - tolerance) <= current_low <= fvg.upper_bound * (1 + tolerance):
                    return fvg
            # 看跌 FVG: 价格反弹到下边界附近
            else:
                if fvg.lower_bound * (1 - tolerance) <= current_high <= fvg.lower_bound * (1 + tolerance):
                    return fvg
        
        return None
    
    def analyze(self, df: pd.DataFrame, **kwargs) -> dict:
        """完整 FVG 分析"""
        all_fvgs = self.detect(df)
        unfilled = self.find_unfilled_fvg(df)
        near_fvg = self.is_price_near_fvg(df)
        
        return {
            "all_fvgs": all_fvgs,
            "unfilled_count": len(unfilled),
            "near_fvg": near_fvg,
            "latest_bullish_fvg": next((f for f in reversed(all_fvgs) if f.direction == "BULLISH"), None),
            "latest_bearish_fvg": next((f for f in reversed(all_fvgs) if f.direction == "BEARISH"), None)
        }

# agent/src/analysis/mtes_v3/layer3/entry_timing.py

import pandas as pd
import numpy as np
from typing import Literal, Optional
from dataclasses import dataclass
from ..base import BaseLayer, EntrySignal
from .fvg_detector import FVGDetector

class EntryTiming(BaseLayer):
    """入场时机检测"""
    
    def __init__(self, rsi_oversold: float = 35, rsi_overbought: float = 65):
        self.rsi_oversold = rsi_oversold
        self.rsi_overbought = rsi_overbought
        self.fvg_detector = FVGDetector()
    
    def validate(self, df: pd.DataFrame) -> bool:
        return len(df) >= 50
    
    def calculate_rsi(self, df: pd.DataFrame, period: int = 14) -> float:
        """计算 RSI"""
        close = df['close']
        delta = close.diff()
        gain = delta.where(delta > 0, 0).rolling(period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(period).mean()
        rs = gain / loss
        rsi = 100 - (100 / (1 + rs))
        return rsi.iloc[-1]
    
    def find_entry(
        self,
        df: pd.DataFrame,
        mtf_bias: str,
        strength_rating: str
    ) -> EntrySignal:
        """寻找入场时机"""
        # 仅在 STRONG/READY 强度时入场
        if strength_rating not in ["STRONG", "READY"]:
            return EntrySignal(
                signal="WAIT",
                reason=f"Weak strength: {strength_rating}"
            )
        
        # 获取 RSI
        rsi = self.calculate_rsi(df)
        
        # 获取 FVG
        fvg_analysis = self.fvg_detector.analyze(df)
        near_fvg = fvg_analysis.get("near_fvg")
        
        # 入场条件判定
        if mtf_bias == "BULL":
            # 多头入场条件: RSI 超卖 + 接近看涨 FVG
            if rsi < self.rsi_oversold and near_fvg and near_fvg.direction == "BULLISH":
                return EntrySignal(
                    signal="LONG",
                    entry_price=near_fvg.upper_bound,
                    stop_loss=near_fvg.lower_bound,
                    reason=f"FVG pullback entry (RSI={rsi:.1f})"
                )
            # 或者: RSI 极度超卖
            elif rsi < 25:
                return EntrySignal(
                    signal="LONG",
                    entry_price=df['close'].iloc[-1],
                    stop_loss=df['low'].rolling(10).min().iloc[-1],
                    reason=f"Extreme oversold (RSI={rsi:.1f})"
                )
        
        elif mtf_bias == "BEAR":
            # 空头入场条件
            if rsi > self.rsi_overbought and near_fvg and near_fvg.direction == "BEARISH":
                return EntrySignal(
                    signal="SHORT",
                    entry_price=near_fvg.lower_bound,
                    stop_loss=near_fvg.upper_bound,
                    reason=f"FVG pullback entry (RSI={rsi:.1f})"
                )
            elif rsi > 75:
                return EntrySignal(
                    signal="SHORT",
                    entry_price=df['close'].iloc[-1],
                    stop_loss=df['high'].rolling(10).max().iloc[-1],
                    reason=f"Extreme overbought (RSI={rsi:.1f})"
                )
        
        return EntrySignal(
            signal="WAIT",
            reason="No entry signal"
        )
    
    def analyze(self, df: pd.DataFrame, **kwargs) -> EntrySignal:
        """完整入场时机分析"""
        mtf_bias = kwargs.get("mtf_bias", "NEUTRAL")
        strength_rating = kwargs.get("strength_rating", "WEAK")
        
        return self.find_entry(df, mtf_bias, strength_rating)
